## 2、去除双细胞
```r
## 使用scDblFinder找双细胞
library(scDblFinder)
library(BiocParallel)
library(Seurat)
SingleCellExperiment_merge = JoinLayers(seurat_use)
SingleCellExperiment_merge = as.SingleCellExperiment(SingleCellExperiment_merge)
result_scDblFinder = scDblFinder(SingleCellExperiment_merge, samples = "sample", BPPARAM = SnowParam(20))

seurat_use@meta.data$scDblFinder.class = result_scDblFinder@colData$scDblFinder.class[match(rownames(seurat_use@meta.data), rownames(result_scDblFinder@colData))]
## 去除双细胞
seurat_use = seurat_use[,which(seurat_use@meta.data$scDblFinder.class=="singlet")]

pdf(paste0(dest_dir, "/去除双细胞后.pdf"), width = 10, height = 5)
```
