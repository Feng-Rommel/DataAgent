## 1、初步找细胞簇
```r
library(Seurat)
## 初步找细胞簇
reduction_use = "integrated.scvi"
number_reduction = 30
seurat_use <- FindNeighbors(object = seurat_use, reduction = reduction_use, dims = 1:number_reduction, k.param = 20)
seu_obj.scvi = seurat_use
seq <- seq(0.1, 1.0, by = 0.1)
for(res in seq){
    seu_obj.scvi <- FindClusters(seu_obj.scvi, resolution = res, n.start = 10, n.iter = 10, algorithm = 4, graph.name = "RNA_snn")
}

library(clustree)
library(patchwork)
library(Seurat)
## 细胞聚簇
p1 <- clustree(seu_obj.scvi, node_label_size = 5, prefix = "RNA_snn_res.") + coord_flip()
pdf(paste0(dest_dir, "/resolution选择.pdf"), width = 8, height = 8)
p1
dev.off()
```

## 4、计算每个簇的基因差异信息
```r
seurat_use = JoinLayers(seurat_use)
markers_positive = FindAllMarkers(seurat_use, only.pos = TRUE,
                                  assay = "SCT")
## 使用log2FC排序
markers_positive = markers_positive[order(markers_positive$avg_log2FC, decreasing = TRUE),]
markers_positive$diff.pct = markers_positive$pct.1 - markers_positive$pct.2
## 使用表达比例差异排序
markers_positive = markers_positive[order(markers_positive$diff.pct, decreasing = TRUE), ]
```

## 5、查看每个簇排名靠前的marker基因
```r
for(cluster in unique(seurat_use@meta.data$seurat_clusters)) {
   markers_positive_tmp = markers_positive[markers_positive$cluster == cluster, ]
   print("cluster", cluster, "的前20个marker基因是: ", rownames(markers_positive_tmp)[1:20])
}
```
