## 样本信息获取
### 一般是从GEO获取，用户指定数据ID(比如GSE159115，需根据实际情况设定)
```r
data_id = "GSE159115" ## 需根据实际情况设置
dest_dir = paste0("./", "2、单细胞数据（scRNA）处理与细胞注释/", data_id, "/")
dir.create(dest_dir)

gset <- getGEO(data_id, destdir="./raw_GEO/",
               AnnotGPL = T,     ## 注释文件
               getGPL = T)       ## 平台文件
gset[[1]]

pd1 <- pData(gset[[1]])#临床信息

library(stringr)
brefInfo = data.frame(sample = rownames(pd1),
                      type1 = pd1$`tissue:ch1`,
                      type2 = pd1$`tumor subtype:ch1`)
brefInfo = brefInfo[-which(brefInfo$type1=="Tumor"&brefInfo$type2=="Chromophobe RCC"),]
brefInfo$group_diagnosis = ifelse(grepl("Benign Adjacent", brefInfo$type1), "control", "disease")
brefInfo$group_diagnosis = factor(brefInfo$group_diagnosis, levels = c("control", "disease"))

write.csv(brefInfo, "./data/brefInfo_GSE159115.csv")
```

## 1、标准的cellranger输出
### 如果是从GEO补充文件下载下来的可能所有样本的数据都在同一目录下，要做如下处理
```r
library(Seurat)
files = list.files(paste0("./raw_sc/", data_id, "_RAW/"))
samples = str_extract(files, "(GSM\\d*)_", group = 1)

for(i in 1:length(unique(samples))){
    dir.create(paste0("./raw_sc/", data_id, "_RAW/", unique(samples)[i]))
}

for(i in 1:length(files)){
    if(grepl("barcodes", files[i])){
        file.rename(paste0("./raw_sc/", data_id, "_RAW/", files[i]), paste0("./raw_sc/", data_id, "_RAW/", samples[i], "/barcodes.tsv.gz"))
    }else if(grepl("features", files[i])){
        file.rename(paste0("./raw_sc/", data_id, "_RAW/", files[i]), paste0("./raw_sc/", data_id, "_RAW/", samples[i], "/features.tsv.gz"))
    }else if(grepl("genes", files[i])){
        file.rename(paste0("./raw_sc/", data_id, "_RAW/", files[i]), paste0("./raw_sc/", data_id, "_RAW/", samples[i], "/features.tsv.gz"))
    }else{
        file.rename(paste0("./raw_sc/", data_id, "_RAW/", files[i]), paste0("./raw_sc/", data_id, "_RAW/", samples[i], "/matrix.mtx.gz"))
    }
}
```

### 将每个样本的文件分别放入一个文件夹下后，可以直接读取
```r
## 读取count数据
samples = brefInfo_analysis$sample
count_all = list()
library(data.table)
library(Seurat)
for(i in 1:length(samples)){
  count_all[[samples[i]]] = Read10X(paste0("./raw_sc/GSE190889_RAW/", samples[i], "/"))
}
lapply(count_all, dim)

## 制作为Seurat对象
library(Seurat)
seurat_samples = list()
for(i in 1:length(count_all)){
  seurat_samples[[samples[i]]] = CreateSeuratObject(counts = count_all[[i]], project = samples[i], min.cells = 3, min.features = 50)
}

samples = brefInfo_analysis$sample
seurat_samples = seurat_samples[samples]
## 合并
seurat_merge = merge(seurat_samples[[1]], seurat_samples[-1], add.cell.ids = samples)
seurat_merge@meta.data$sample = seurat_merge@meta.data$orig.ident
seurat_merge@meta.data$group_diagnosis = brefInfo_analysis$group_diagnosis[match(seurat_merge@meta.data$sample, brefInfo_analysis$sample)]
```
