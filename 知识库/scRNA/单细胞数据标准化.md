s.genes <- cc.genes$s.genes
g2m.genes <- cc.genes$g2m.genes

# 如果是小鼠数据，建议使用 tolower() 或转换为小鼠基因名

# 计算细胞周期评分

head(seurat_use@meta.data)
# 找出变异最大的2000个基因
seurat_use <- FindVariableFeatures(seurat_use, 
                                   selection.method = "vst",
                                   nfeatures = 2000, 
                                   verbose = FALSE)
# 缩放计数
seurat_use <- ScaleData(seurat_use)
# 运行 PCA
seurat_use <- RunPCA(seurat_use)

# 使用细胞周期阶段对PCA进行着色
DimPlot(seurat_use,
        reduction = "pca",
        group.by= "Phase",
        split.by = "Phase")

        reduction = "pca",
        group.by= "group_diagnosis",
        split.by = "group_diagnosis")
